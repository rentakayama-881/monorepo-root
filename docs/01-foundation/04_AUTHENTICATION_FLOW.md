```javascript
// lib/tokenRefresh.js
export async function getValidToken() {
  const token = getToken();
  if (!token) return null;
  
  // Check if token is about to expire
  if (isTokenExpiringSoon(token)) {
    try {
      const newToken = await refreshToken();
      saveToken(newToken);
      return newToken;
    } catch {
      clearToken();
      return null;
    }
  }
  
  return token;
}
```
